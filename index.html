Recursion — Base Cases
# Factorial
def factorial(n):
    if n == 0: return 1
    return n * factorial(n-1)

# Sum of digits
def sum_dig(n):
    if n < 10: return n
    return n%10 + sum_dig(n//10)
# Fibonacci
def fib(n):
    if n <= 1: return n
    return fib(n-1) + fib(n-2)

# Power
def power(b, e):
    if e == 0: return 1
    return b * power(b, e-1)
Lambda · map · filter
# lambda: anonymous one-liner
sq = lambda x: x**2
add = lambda a, b: a + b

# map: apply function to every element
list(map(lambda x: x**2, [1,2,3]))      # [1,4,9]
list(map(lambda c: c*9/5+32, celsius))   # C→F

# filter: keep elements where function is True
list(filter(lambda x: x%2==0, nums))     # evens only
list(filter(lambda s: s==s[::-1], words)) # palindromes
Closure · Memoization · Decorator
# Closure
def make_counter():
    count = 0
    def increment():
        nonlocal count
        count += 1
        print(count)
    return increment

# Memoization
def memoize(f):
    cache = {}
    def wrapper(n):
        if n not in cache:
            cache[n] = f(n)
        return cache[n]
    return wrapper

# Decorator (@timer)
import time
def timer(f):
    def wrapper(*args):
        start = time.time()
        result = f(*args)
        print(f"{time.time()-start:.4f}s")
        return result
    return wrapper

Core Patterns
# Write
with open("f.txt", "w") as f:
    f.write("hello\n")
    f.writelines(["a\n", "b\n"])

# Read
with open("f.txt", "r") as f:
    content = f.read()          # whole file → str
    lines   = f.readlines()     # → list of lines
    line    = f.readline()      # → one line

# Iterate line by line (memory efficient)
with open("f.txt") as f:
    for line in f:
        print(line.strip())

# Append
with open("f.txt", "a") as f:
    f.write("new line\n")
seek() & tell()
f.tell()       # current position (bytes from start)
f.seek(0)      # go to start
f.seek(0, 2)   # go to end  (2 = SEEK_END)
f.seek(5)      # jump to byte 5
Delete a specific line
with open("f.txt") as f:
    lines = f.readlines()
lines.pop(n-1)               # delete line n (1-indexed)
with open("f.txt", "w") as f:
    f.writelines(lines)
Always use with open(...) — auto-closes file even on error. Never use f.close() manually in exam code.
